CREATE DATABASE StoreDatabase;

CREATE TABLE Product (
  ProductID int not null primary key,
  Name char(30) not null,
  Quantity int,
  Price decimal(9,2),
  Provider char(30),
  ProviderContact char(12));

CREATE TABLE Order (
  OrderID int not null primary key,
  Total decimal(9,2),
  CustomerID int,
  EmployeeID int,
  Date datetime);
